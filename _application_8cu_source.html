133 
134 #if THRUST_DEVICE_SYSTEM == THRUST_DEVICE_SYSTEM_CUDA
135  int deviceCount;
136  cudaGetDeviceCount(&deviceCount);
137 
138  // Get Batch system number of nodes if possible
139  auto PBS_NUM_NODES = getenv("PBS_NUM_NODES");
140 
141  int nodes = PBS_NUM_NODES == nullptr ? 1 : atoi(PBS_NUM_NODES);
142 
143  if(nodes == 0)
144  nodes = 1;
145 
146  int procsPerNode = numProcs / nodes;
147  int localRank = myId % procsPerNode;
148 
149  // Note, this will (probably) be overwritten by gpu-set-device calls...
150  if(deviceCount == 1 && localRank > 1) {
151  // Multi-process to one GPU!
152  gpuDev_ = 0;
153  } else if(procsPerNode > 1 && deviceCount > 1) {
154  if(localRank <= deviceCount) {
155  // setting multiple processes to multiple GPU's
156  gpuDev_ = localRank;
157  } else {
158  // More multiple processes than GPU's, distribute sort of evenly!
159  gpuDev_ = localRank % deviceCount;
160  }
161  } else {
162  // multiple GPU's, using one process
163  gpuDev_ = 0;
164  }
165 
166  std::cout << "MPI using CUDA device: " << gpuDev_ << std::endl;
167  cudaSetDevice(gpuDev_);
168 #endif
169 #endif
170 
171  // Fallthrough is useful for most models of GooFit subcommands
172  fallthrough();
173

175 #ifndef GOOFIT_MPI
176  add_option("--gpu-dev", gpuDev_, "GPU device to use", true)->group("GooFit");

240  // Run from current directory

254  }
255
