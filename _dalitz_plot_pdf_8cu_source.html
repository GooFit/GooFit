315 EXEC_TARGET devcomplex<fptype> SpecialResonanceIntegrator::operator()(thrust::tuple<int, fptype*> t) const {
316  // Bin index, base address [lower, upper, numbins]
317  // Notice that this is basically MetricTaker::operator (binned) with the special-case knowledge
318  // that event size is two, and that the function to call is dev_DalitzPlot_calcIntegrals.
319 
320  int globalBinNumber = thrust::get<0>(t);
321  fptype lowerBoundM12 = thrust::get<1>(t)[0];
322  fptype upperBoundM12 = thrust::get<1>(t)[1];
323  int numBinsM12 = (int) FLOOR(thrust::get<1>(t)[2] + 0.5);
324  int binNumberM12 = globalBinNumber % numBinsM12;
325  fptype binCenterM12 = upperBoundM12 - lowerBoundM12;
326  binCenterM12 /= numBinsM12;
327  binCenterM12 *= (binNumberM12 + 0.5);
328  binCenterM12 += lowerBoundM12;
329 
330  globalBinNumber /= numBinsM12;
331  fptype lowerBoundM13 = thrust::get<1>(t)[3];
332  fptype upperBoundM13 = thrust::get<1>(t)[4];
333  int numBinsM13 = (int) FLOOR(thrust::get<1>(t)[5] + 0.5);
334  fptype binCenterM13 = upperBoundM13 - lowerBoundM13;
335  binCenterM13 /= numBinsM13;
336  binCenterM13 *= (globalBinNumber + 0.5);
337  binCenterM13 += lowerBoundM13;
338 
339  unsigned int* indices = paramIndices + parameters;
340  devcomplex<fptype> ret = device_DalitzPlot_calcIntegrals(binCenterM12, binCenterM13, resonance_i, resonance_j,
341  cudaArray, indices);
342 
343  fptype fakeEvt[10]; // Need room for many observables in case m12 or m13 were assigned a high index in an event-weighted fit.
344  fakeEvt[indices[indices[0] + 2 + 0]] = binCenterM12;
345  fakeEvt[indices[indices[0] + 2 + 1]] = binCenterM13;
346  unsigned int numResonances = indices[2];
347  int effFunctionIdx = parIndexFromResIndex_DP(numResonances);
348  fptype eff = callFunction(fakeEvt, indices[effFunctionIdx], indices[effFunctionIdx + 1]);
349 
350  // Multiplication by eff, not sqrt(eff), is correct:
351  // These complex numbers will not be squared when they
352  // go into the integrals. They've been squared already,
353  // as it were.
354  ret *= eff;
355  // printf("ret %f %f %f %f %f\n",binCenterM12, binCenterM13, ret.real, ret.imag, eff );
356  return ret;
357 }
