39 __device__ fptype BL_PRIME(fptype z2, fptype z02, int L) {
40  if(0 == L)
41  return 1.0;
42  else if(1 == L)
43  return (1 + z02) / (1 + z2);
44  else if(2 == L)
45  return (z02 * z02 + 3 * z02 + 9) / (z2 * z2 + 3 * z2 + 9);
46  else {
47  printf("ERROR! Oribtal > 2 not supported!\n");
48  return 0;
49  }
50 
51  // Spin 3 and up not accounted for.
52 }

87 __device__ fpcomplex BW(fptype Mpair, fptype m1, fptype m2, unsigned int *indices) {
88  fptype meson_radius = functorConstants[indices[7]];
89  fptype resmass = cudaArray[indices[2]];
90  fptype reswidth = cudaArray[indices[3]];
91  unsigned int orbital = indices[4];
92  unsigned int FF = indices[6];
93 
94  const unsigned int to2Lplus1 = 2 * orbital + 1;
95 
96  fptype mass = resmass;
97  fptype width = reswidth;
98  fptype mumsRecoMass2 = Mpair * Mpair;
99 
100  fptype mpsq = (m1 + m2) * (m1 + m2);
101  fptype mmsq = (m1 - m2) * (m1 - m2);
102  fptype num = (mumsRecoMass2 - mpsq) * (mumsRecoMass2 - mmsq);
103  fptype num2 = (mass * mass - mpsq) * (mass * mass - mmsq);
104  fptype pABSq = num / (4 * mumsRecoMass2);
105  fptype prSqForGofM = num2 / (4 * mass * mass);
106  fptype prSq2 = prSqForGofM < 0 ? 0 : prSqForGofM;
107  prSqForGofM = fabs(prSqForGofM);
108 
109  fptype pratio = sqrt(pABSq / prSqForGofM);
110 
111  fptype pratio_to_2Jplus1 = 1;
112 
113  for(int i = 0; i < to2Lplus1; i++) {
114  pratio_to_2Jplus1 *= pratio;
115  }
116 
117  fptype mratio = mass / Mpair;
118  fptype r2 = meson_radius * meson_radius;
119  fptype thisFR = BL_PRIME(pABSq * r2, prSqForGofM * r2, orbital);
120  fptype frFactor = 1;
121 
122  if(0 != orbital && 0 != FF) {
123  frFactor = (FF == 1 ? BL(pABSq * r2, orbital) : BL_PRIME(pABSq * r2, prSq2 * r2, orbital));
124  frFactor = (FF == 3 ? BL2(pABSq * r2, orbital) : frFactor);
125  }
126 
127  fptype GofM = width * pratio_to_2Jplus1 * mratio * thisFR;
128 
129  fptype gamma = mass * sqrt((mass * mass + width * width));
130  fptype k = (2.0 * sqrt(2.0) / M_PI) * mass * width * gamma
131  / sqrt(mass * mass + gamma); // Note added additional factor of 2*sqrt(2)/PI here so results are
132  // comparable to MINT3. MINT2 doesn't have include this.
133 
134  fpcomplex BW(mass * mass - mumsRecoMass2, mass * GofM);
135  fptype den = (mass * mass - mumsRecoMass2) * (mass * mass - mumsRecoMass2) + mass * GofM * mass * GofM;
136 
137  fpcomplex ret = (sqrt(k * frFactor)) / den * BW;
138  // printf("m1, m2, Mpair, to2Lplus1, GofM, thisFR, pratio, mratio, pABSq , prSqForGofM, FF, ret.real, ret.imag\n");
139  // printf("BW %.7g, %.7g, %.7g, %i, %i, %i, %i\n",meson_radius, resmass, reswidth, orbital, FF, indices[2],
140  // indices[3]);
141  // printf("BW %.7g, %.7g, %.7g, %i, %.7g, %.7g, %.7g, %.7g, %.7g, %.7g, %.7g, %.7g, %.7g\n", m1, m2, Mpair,
142  // to2Lplus1, GofM, thisFR, pratio, mratio, pABSq, prSqForGofM, frFactor, ret.real, ret.imag );
143  return ret;
144 }

198 __device__ fpcomplex bugg_rho2(const fptype &s, const fptype m) {
199  fptype rho_squared = 1. - 4. * m * m / s;
200  fpcomplex returnVal = (rho_squared >= 0) ? fpcomplex(1, 0) : fpcomplex(0, 1);
201  rho_squared = (rho_squared >= 0) ? sqrt(rho_squared) : sqrt(-rho_squared);
202  return rho_squared * returnVal;

226 __device__ fpcomplex bugg_MINT(fptype Mpair, fptype m1, fptype m2, unsigned int *indices) {
227  fptype s = Mpair * Mpair;
228 
229  fptype M = 0.953;
230  fptype b1 = 1.302;
231  fptype b2 = 0.340;
232  fptype A = 2.426;
233  fptype g_4pi = 0.011;
234  fptype g_2K = 0.6;
235  fptype g_2eta = 0.2;
236  fptype alpha = 1.3;
237  fptype sA = 0.41;
238  fptype s0_4pi = 7.082 / 2.845;
239  fptype lambda_4pi = 2.845;
240 
241  fptype g1sq = (b1 + b2 * s) * exp(-(s - M * M) / A);
242  fptype z = bugg_j1(s, mPiPlus) - bugg_j1(M * M, mPiPlus);
243 
244  fpcomplex gamma_2pi = fpcomplex(
245  g1sq * (s - sA * mPiPlus * mPiPlus) / (M * M - sA * mPiPlus * mPiPlus) * bugg_rho2(s, mPiPlus).real(), 0);
246  fpcomplex gamma_2K = g_2K * g1sq * s / (M * M)
247  * exp((-1) * alpha * sqrt((s - 4. * mKPlus * mKPlus) * (s - 4. * mKPlus * mKPlus)))
248  * bugg_rho2(s, mKPlus);
249  fpcomplex gamma_2eta = g_2eta * g1sq * s / (M * M)
250  * exp((-1) * alpha * sqrt((s - 4. * mEta * mEta) * (s - 4. * mEta * mEta)))
251  * bugg_rho2(s, mEta);
252  fpcomplex gamma_4pi = fpcomplex(bugg_Gamma_4pi(s, mPiPlus, g_4pi, M, lambda_4pi, s0_4pi), 0);
253 
254  fpcomplex Gamma_tot = gamma_2pi + gamma_2K + gamma_2eta + gamma_4pi;
255 
256  // fpcomplex num = M * gamma_2pi; //only for elastic scattering, not production
257  fpcomplex den
258  = fpcomplex(M * M - s - M * g1sq * (s - sA * mPiPlus * mPiPlus) / (M * M - sA * mPiPlus * mPiPlus) * z, 0)
259  - fpcomplex(0, 1) * M * Gamma_tot;
260  fpcomplex returnVal = 1.0 / den;
261  // printf("Bugg %.5g %.5g %.5g %.5g %.5g %.5g %.5g %.5g \n",gamma_2pi.real, gamma_2pi.imag, gamma_2K.real,
262  // gamma_2K.imag, gamma_2eta.real, gamma_2eta.imag, gamma_4pi.real, gamma_4pi.imag);
263  // printf("Bugg %.5g %.5g %.5g %.5g %.5g %.5g %.5g %.5g %.5g \n",Mpair, Gamma_tot.real, Gamma_tot.imag, g1sq, z,
264  // den.real, den.imag, returnVal.real, returnVal.imag);
265 
266  // the factor sqrt(1000) gives the correct result in comparison with mint2, I think its because BW/SBW
267  // have a factor of sqrt(k) which these lineshapes dont have. For now this stays because it works. further
268  // investigation needed.
269  return returnVal * sqrt(1000.0);
270 }

297  fpcomplex gamma_2eta = g_2eta * g1sq * s / (M * M) * exp((-1) * alpha * tmp2) * bugg_rho2(s, mEta);

301 
302  // fpcomplex num = M * gamma_2pi; //only for elastic scattering, not production
303  fpcomplex den = fpcomplex(M * M - s - adlerZero * g1sq * z, 0) - fpcomplex(0, 1) * Gamma_tot;

308  // den.real, den.imag, returnVal.real, returnVal.imag);
309 
310  return returnVal;
311 }

460 __device__ fptype getSpline(fptype x, bool continued, unsigned int *indices) {
461  const fptype s_min = GOOFIT_GET_CONST(8);
462  const fptype s_max = GOOFIT_GET_CONST(9);
463  const unsigned int nBins = GOOFIT_GET_INT(10);
464 
465  // 11 is the first spine knot, 11+nBins is the first curvature
466 
467  if(x <= s_min)
468  return continued ? GOOFIT_GET_PARAM(11 + 0) : 0;
469  if(x >= s_max)
470  return continued ? GOOFIT_GET_PARAM(11 + nBins - 1) : 0;
471 
472  fptype spacing = (s_max - s_min) / (nBins - 1.);
473  fptype dx = fmod((x - s_min), spacing);
474 
475  auto bin = static_cast<unsigned int>((x - s_min) / spacing);
476 
477  fptype m_x_0 = GOOFIT_GET_PARAM(11 + bin);
478  fptype m_x_1 = GOOFIT_GET_PARAM(11 + bin + 1);
479  fptype m_xf_0 = GOOFIT_GET_CONST(11 + bin + nBins);
480  fptype m_xf_1 = GOOFIT_GET_CONST(11 + bin + nBins + 1);
481 
482  return m_x_0 + dx * ((m_x_1 - m_x_0) / spacing - (m_xf_1 + 2 * m_xf_0) * spacing / 6) + dx * dx * m_xf_0
483  + dx * dx * dx * (m_xf_1 - m_xf_0) / (6 * spacing);
484 }

486 __device__ fptype kFactor(fptype mass, fptype width) {
487  fptype gamma = mass * sqrt(POW2(mass) + POW2(width));
488  fptype k = 2 * sqrt(2.) * mass * width * gamma / (M_PI * sqrt(POW2(mass) + gamma));
489  return sqrt(k);

493  const fptype mass = GOOFIT_GET_PARAM(2);
494  const fptype width = GOOFIT_GET_PARAM(3);

496  const fptype radius = GOOFIT_GET_CONST(7);

515  fpcomplex iBW = fpcomplex(POW2(mass) - s, -mass * running_width);
516  return norm / iBW;

589 __device__ fpcomplex kMatrixFunction(fptype Mpair, fptype m1, fptype m2, unsigned int *indices) {
590  // const fptype mass = GOOFIT_GET_PARAM(2);
591  // const fptype width = GOOFIT_GET_PARAM(3);
592  // const unsigned int L = GOOFIT_GET_INT(4);
593  // const fptype radius = GOOFIT_GET_CONST(7);
594 
595  // const fptype pTerm = GOOFIT_GET_INT();
596 
597  unsigned int pterm = GOOFIT_GET_INT(8);
598  bool is_pole = GOOFIT_GET_INT(9) == 1;
599 
600  fptype sA0 = GOOFIT_GET_PARAM(10);
601  fptype sA = GOOFIT_GET_PARAM(11);
602  fptype s0_prod = GOOFIT_GET_PARAM(12);
603  fptype s0_scatt = GOOFIT_GET_PARAM(13);
604 
605  Eigen::Array<fptype, NCHANNELS, 1> fscat;
606  Eigen::Array<fptype, NPOLES, 1> pmasses;
607  Eigen::Array<fptype, NPOLES, NPOLES> couplings;
608 
609  for(int i = 0; i < NCHANNELS; i++) {
610  fscat(i) = GOOFIT_GET_PARAM(14 + i);
611  }
612 
613  for(int i = 0; i < NPOLES; i++) {
614  for(int j = 0; j < NPOLES; j++)
615  couplings(i, j) = GOOFIT_GET_PARAM(14 + NCHANNELS + i * (NPOLES + 1) + j);
616  pmasses(i) = GOOFIT_GET_PARAM(14 + NCHANNELS + i * (NPOLES + 1) + NPOLES);
617  }
618 
619  fptype s = POW2(Mpair);
620 
621  // constructKMatrix
622 
623  Eigen::Array<fptype, NCHANNELS, NCHANNELS> kMatrix;
624  kMatrix.setZero();
625 
626  // TODO: Make sure the order (k,i,j) is correct
627 
628  for(int i = 0; i < 5; i++) {
629  for(int j = 0; j < 5; j++) {
630  for(int k = 0; k < 5; k++)
631  kMatrix(i, j) += couplings(k, i) * couplings(k, j) / (pmasses(k) - s);
632  if(i == 0 || j == 0) // Scattering term
633  kMatrix(i, j) += fscat(i + j) * (1 - s0_scatt) / (s - s0_scatt);
634  }
635  }
636 
637  fptype adlerTerm = (1. - sA0) * (s - sA * mPiPlus * mPiPlus / 2) / (s - sA0);
638 
639  Eigen::Matrix<fptype, 5, 1> phaseSpace;
640  phaseSpace << phsp_twoBody(s, mPiPlus, mPiPlus), phsp_twoBody(s, mKPlus, mKPlus), phsp_fourPi(s),
641  phsp_twoBody(s, mEta, mEta), phsp_twoBody(s, mEta, mEtap);
642 
643  Eigen::Array<fpcomplex, NCHANNELS, NCHANNELS> F = getPropagator(kMatrix, phaseSpace, adlerTerm);
644 
645  if(is_pole) { // pole
646  fpcomplex M = 0;
647  for(int i = 0; i < NCHANNELS; i++) {
648  fptype pole = couplings(i, pterm);
649  M += F(0, i) * pole;
650  }
651  return M / (POW2(pmasses(pterm)) - s);
652  } else { // prod
653  return F(0, pterm) * (1 - s0_prod) / (s - s0_prod);
654  }
655 }

658  fptype mp = (m0 + m1);
659  fptype mm = (m0 - m1);
660  fptype a2 = (1.0 - mp * mp / s) * (1.0 - mm * mm / s);

664 __device__ fpcomplex FOCUSFunction(fptype Mpair, fptype m1, fptype m2, unsigned int *indices) {
665  fptype s = POW2(Mpair);
666  unsigned int mod = GOOFIT_GET_INT(8);
667 
668  // mKPlus, mPiPlus, mEtap
669  constexpr fptype sNorm = mKPlus * mKPlus + mPiPlus * mPiPlus;
670  constexpr fptype s12 = 0.23;
671  constexpr fptype s32 = .27;
672 
673  fptype I12_adler = (s - s12) / sNorm;
674  fptype I32_adler = (s - s32) / sNorm;
675 
676  fptype rho1 = phsp_FOCUS(s, mKPlus, mPiPlus);
677  fptype rho2 = phsp_FOCUS(s, mKPlus, mEtap);
678 
679  fptype pmass = 1.7919;
680  Eigen::Array<fptype, 2, 1> coupling;
681  coupling << 0.31072, -0.02323;
682  fptype X = s / sNorm - 1;
683 
684  // constructKMatrix
685  Eigen::Array<fptype, 2, 2> kMatrix;
686 
687  kMatrix(0, 0) = coupling(0) * coupling(0) / (pmass - s) + 0.79299 - 0.15099 * X + 0.00811 * POW2(X);
688  kMatrix(1, 1) = coupling(1) * coupling(1) / (pmass - s) + 0.17054 - 0.0219 * X + 0.00085655 * POW2(X);
689  kMatrix(1, 0) = coupling(1) * coupling(0) / (pmass - s) + 0.15040 - 0.038266 * X + 0.0022596 * POW2(X);
690  kMatrix(0, 1) = coupling(0) * coupling(1) / (pmass - s) + 0.15040 - 0.038266 * X + 0.0022596 * POW2(X);
691 
692  fptype K11 = I12_adler * kMatrix(0, 0);
693  fptype K12 = I12_adler * kMatrix(0, 1);
694  fptype K22 = I12_adler * kMatrix(1, 1);
695 
696  fptype K32 = I32_adler * (-0.22147 + 0.026637 * X - 0.00092057 * POW2(X));
697 
698  fptype detK = K11 * K22 - K12 * K12;
699  fpcomplex del{1 - rho1 * rho2 * detK, -(rho1 * K11 + rho2 * K22)};
700 
701  fpcomplex T11{1., -rho2 * K22};
702  fpcomplex T22{1., -rho1 * K11};
703  fpcomplex T12{0., rho2 * K12};
704 
705  fpcomplex T32 = 1. / fpcomplex(1, -K32 * rho1);
706 
707  if(mod == static_cast<unsigned int>(Lineshapes::FOCUS::Mod::Kpi))
708  return fpcomplex(K11, -rho2 * detK) / del;
709  else if(mod == static_cast<unsigned int>(Lineshapes::FOCUS::Mod::KEta))
710  return K12 / del;
711  else /*if(mod==Lineshapes::FOCUS::Mod::I32)*/
712  return T32;
713 
714  return {0., 0.};
715 }

739 std::vector<fptype> make_spline_curvatures(std::vector<Variable> vars, Lineshapes::spline_t SplineInfo) {
740  size_t size = std::get<2>(SplineInfo) - 2;
741  Eigen::Matrix<fptype, Eigen::Dynamic, Eigen::Dynamic> m(size, size);
742  for(size_t i = 0; i < size; i++) {
743  m(i, i) = 4;
744  if(i != size - 1) {
745  m(i, i + 1) = 1;
746  m(i + 1, 1) = 1;
747  }
748  }
749  m = m.inverse();
750 
751  Eigen::Matrix<fptype, Eigen::Dynamic, 1> L(size);
752  for(unsigned int i = 0; i < size; ++i)
753  L[i] = vars[i + 2].getValue() - 2 * vars[i + 1].getValue() + vars[i].getValue();
754 
755  auto mtv = m * L;
756  fptype spacing = (std::get<0>(SplineInfo) - std::get<1>(SplineInfo)) / std::get<2>(SplineInfo);
757 
758  std::vector<fptype> ret(vars.size(), 0);
759  for(unsigned int i = 0; i < size; ++i) {
760  ret.at(i + 1) = 6 * mtv(i) / POW2(spacing);
761  }
762  return ret;
763 }
