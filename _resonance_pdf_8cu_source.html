9 __device__ fptype twoBodyCMmom(double rMassSq, fptype d1m, fptype d2m) {
10  // For A -> B + C, calculate momentum of B and C in rest frame of A.
11  // PDG 38.16.
12  fptype kin1 = 1 - POW2(d1m + d2m) / rMassSq;
13 
14  if(kin1 >= 0)
15  kin1 = sqrt(kin1);
16  else
17  kin1 = 1;
18 
19  fptype kin2 = 1 - POW2(d1m - d2m) / rMassSq;
20 
21  if(kin2 >= 0)
22  kin2 = sqrt(kin2);
23  else
24  kin2 = 1;
25 
26  return 0.5 * sqrt(rMassSq) * kin1 * kin2;
27 }

446  dmKK = mKKlimits[khiAB] - mKKlimits[kloAB];
447  aa = (mKKlimits[khiAB] - mAB) / dmKK;
448  bb = 1 - aa;
449  aa3 = aa * aa * aa;
450  bb3 = bb * bb * bb;
451  // ret += aa * pwa_coefs[kloAB] + bb * pwa_coefs[khiAB] + ((aa3 - aa)*pwa_coefs_prime[kloAB] + (bb3 - bb) *
452  // pwa_coefs_prime[khiAB]) * (dmKK*dmKK)/6.0;
453  ret.real(ret.real() + aa * pwa_coefs_real_kloAB + bb * pwa_coefs_real_khiAB
454  + ((aa3 - aa) * pwa_coefs_prime_real_kloAB + (bb3 - bb) * pwa_coefs_prime_real_khiAB) * (dmKK * dmKK)
455  / 6.0);
456  ret.imag(ret.imag() + aa * pwa_coefs_imag_kloAB + bb * pwa_coefs_imag_khiAB
457  + ((aa3 - aa) * pwa_coefs_prime_imag_kloAB + (bb3 - bb) * pwa_coefs_prime_imag_khiAB) * (dmKK * dmKK)
458  / 6.0);
459  khiAB = khiAC;
460  mAB = mAC;
461  }
462  return ret;
