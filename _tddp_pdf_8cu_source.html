797 EXEC_TARGET WaveHolder_s SpecialWaveCalculator::operator()(thrust::tuple<int, fptype*, int> t) const {
798  // Calculates the BW values for a specific resonance.
799  // The 'A' wave stores the value at each point, the 'B'
800  // at the opposite (reversed) point.
801 
802  WaveHolder_s ret;

806  ret.bi_imag = 0.0;
807 
808  int evtNum = thrust::get<0>(t);
809  fptype* evt = thrust::get<1>(t) + (evtNum * thrust::get<2>(t));
810 
811  unsigned int* indices = paramIndices + parameters; // Jump to TDDP position within parameters array
812  fptype m12 = RO_CACHE(evt[indices[4 + indices[0]]]);
813  fptype m13 = RO_CACHE(evt[indices[5 + indices[0]]]);
814 
815  fptype motherMass = functorConstants[indices[1] + 0];
816  fptype daug1Mass = functorConstants[indices[1] + 1];

819 
820  if(!inDalitz(m12, m13, motherMass, daug1Mass, daug2Mass, daug3Mass))
821  return ret;
822 
823  fptype m23 = motherMass*motherMass + daug1Mass*daug1Mass + daug2Mass*daug2Mass + daug3Mass*daug3Mass - m12 - m13;
824 
825  int parameter_i = parIndexFromResIndex(resonance_i); // Find position of this resonance relative to TDDP start
