173 double calcGauInteg(double x1, double x2) {
174  if(x1 > x2) {
175  double swp = x2;
176  x2 = x1;
177  x1 = swp;
178  }
179 
180  double sum1 = x1, sum2 = x2;

186  value2 = value2*x2*x2/(2*i+1);

247 bool cpuDalitz(fptype m12, fptype m13, fptype bigM, fptype dm1, fptype dm2, fptype dm3) {
248  if(m12 < POW(dm1 + dm2, 2))
249  return false; // This m12 cannot exist, it's less than the square of the (1,2) particle mass.
250 
251  if(m12 > POW(bigM - dm3, 2))
252  return false; // This doesn't work either, there's no room for an at-rest 3 daughter.
253 
254  // Calculate energies of 1 and 3 particles in m12 rest frame.
255  fptype e1star = 0.5 * (m12 - dm2*dm2 + dm1*dm1) / SQRT(m12);
256  fptype e3star = 0.5 * (bigM*bigM - m12 - dm3*dm3) / SQRT(m12);
257 
258  // Bounds for m13 at this value of m12.
259  fptype minimum = POW(e1star + e3star, 2) - POW(SQRT(e1star*e1star - dm1*dm1) + SQRT(e3star*e3star - dm3*dm3), 2);
260 
261  if(m13 < minimum)
262  return false;
263 
264  fptype maximum = POW(e1star + e3star, 2) - POW(SQRT(e1star*e1star - dm1*dm1) - SQRT(e3star*e3star - dm3*dm3), 2);
265 
266  if(m13 > maximum)
267  return false;
268 
269  return true;
270 }
