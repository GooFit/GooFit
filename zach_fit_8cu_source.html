32 TH1D *getMCData(DataSet *data, Observable var, std::string filename) {
33  TH1D *mchist = new TH1D{"mc_hist", "", 300, 0.1365, 0.1665};
34  std::ifstream mcreader{filename};
35 
36  while(mcreader >> var) {
37  if(!var)
38  continue;
39  data->addEvent();
40  mchist->Fill(var.getValue());
41  }
42 
43  mchist->SetStats(false);
44  mchist->SetMarkerStyle(8);
45  mchist->SetMarkerSize(0.6);
46 
47  GOOFIT_INFO("MC events: {}", data->getNumEvents());
48  if(data->getNumEvents() == 0)
49  throw GooFit::GeneralError("No MC events read in!");
50  return mchist;
51 }
